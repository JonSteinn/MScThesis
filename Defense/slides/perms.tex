\begin{frame}{Permutation}
    \begin{block}{Definition}<1->
    A \emph{permutation} $\pi$ is a bijection between a set and itself.
    \end{block}
    \begin{block}{Example}<2->
        The permutations of size $3$ are
        \[
            123, 132, 213, 231, 312, 321.
        \]
    \end{block}
